# Copy these environment variables to your .env.local file

# Vercel Blob Storage Token
# Get this from: https://vercel.com/dashboard/stores
# 1. Go to your Vercel dashboard
# 2. Click on "Storage" tab
# 3. Create a new Blob store or use existing one
# 4. Copy the "BLOB_READ_WRITE_TOKEN"
BLOB_READ_WRITE_TOKEN=your_vercel_blob_token_here

# Instructions:
# 1. Copy the line above to your .env.local file
# 2. Replace "your_vercel_blob_token_here" with your actual token
# 3. Make sure .env.local is in your .gitignore (it should be by default)

# To get your token:
# - Visit: https://vercel.com/dashboard/stores
# - Create a new Blob store if you don't have one
# - Copy the BLOB_READ_WRITE_TOKEN value
